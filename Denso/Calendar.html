def self.load(type = :production)
  content = Net::HTTP.get(uri)
  doc = Nokogiri::HTML(content)
  tables = doc.xpath("//h2/a[@id='#{type}']/../../following-sibling::div[contains(@class, 'denso-calendar')][1]")

  Calendar.new(tables)
end
